338 void Add(const ICallable<TRet(Args...)> &callable)
339 {
340 // 当添加的可调用对象与当前委托类型相同时（针对单播委托进行优化）：
341 // - 若委托内容为空，则直接返回
342 // - 若委托内容只有一个元素，则克隆该元素并添加到当前委托中
343 // - 否则，直接添加该可调用对象的克隆
344 if (callable.GetTypeInfo() == GetTypeInfo()) {
345 auto &delegate = static_cast<const Delegate &>(callable);
346 if (delegate._data.empty()) {
347 return;
348 } else if (delegate._data.size() == 1) {
349 _data.emplace_back(delegate._data.front()->Clone());

351 }
352 }
353 _data.emplace_back(callable.Clone());
